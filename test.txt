
A heap is a specialized tree-based data structure that satisfies the heap property. Heaps are typically used to efficiently manage and organize data with specific characteristics, such as finding the maximum or minimum element quickly. There are two main types of heaps: the Max-Heap and the Min-Heap.

const pivotIndex = (newArr,start=0,end=arr.length)=>{
    let pivotEle = newArr[start];
    let swap_index = start;
    for(let i=start + 1;i<end;i++){
        if(pivotEle > newArr[i]){
            swap_index++;
            [newArr[i],newArr[swap_index]] = [newArr[swap_index],newArr[i]];
        }
    }
    [newArr[start],newArr[swap_index]] = [newArr[swap_index],newArr[start]];
    return swap_index;
}

const QuickSort = (arr,left=0,right=arr.length-1)=>{
    if(left < right){
        const pivot_index = pivotIndex(arr,left,right+1);
        QuickSort(arr,left,pivot_index-1);
        QuickSort(arr,pivot_index+1,right);
    }
    return arr;
}
